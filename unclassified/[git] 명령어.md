# git 명령어

```bash
# 유저 이름 등록
$ git config --global user.name gghotted

# 유저 메일 등록
$ git config --global user.email gghotted2@naver.com

# 초기화
$ git init

# add 특정 파일 
$ git add git.md

# 특정 폴더
$ git add images/

# 현재 디렉토리
$ git add .

# 파일 상태
$ git status

# 커밋
$ git commit -m '마크다운 및 git 정리'

# 원격 저장소 등록
$ git remote add origin 깃허브url

# 푸쉬
$ git push origin master

# 풀
$ git pull origin master

# 클론
$ git clone 깃허브url
```
